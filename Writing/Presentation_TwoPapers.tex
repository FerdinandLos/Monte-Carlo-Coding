\section{Introduction}

\begin{frame}{The central problem}
  \begin{itemize}
    \item Structural VARs are widely used for macroeconomic inference.
    \item Impulse responses depend strongly on the estimated lag structure.
    \item In finite samples, lag-order choices are uncertain and can be highly influential.
    \item Set-identified SVARs add another layer of uncertainty through the identification step.
  \end{itemize}
  \vspace{0.4cm}
  \begin{block}{Research question}
    Which estimation strategy recovers the true impulse response function most reliably when the lag order is uncertain?
  \end{block}
\end{frame}

\section{Theory}

% Placeholder: joint theory (VAR framework, sign restrictions, information criteria,
% then BVAR for Akin / BMA for Ferdinand). Content to be aligned with the papers.
\begin{frame}{Theory: overview}
  \begin{itemize}
    \item VAR framework and impulse response functions.
    \item Set identification via sign restrictions and the median target.
    \item Information criteria and the bias--variance trade-off.
    \item Bayesian extension: shrinkage (Paper I) and model averaging (Paper II).
  \end{itemize}
\end{frame}

% ===== T1: VAR framework -> IRF, and the two channels the IRF depends on =====
\begin{frame}{From VAR to impulse responses}
  \textbf{A reduced-form VAR explains today by its own past:}
  \[
    Y_t = v + A_1 Y_{t-1} + \dots + A_{p_0} Y_{t-p_0} + e_t,
    \qquad \mathbb{E}[e_t e_t'] = \Sigma
  \]
  Inverting it (infinite MA form) rewrites today as accumulated past shocks:
  \[
    Y_t = \mu + \sum_{i=0}^{\infty} \Phi_i\, e_{t-i}
  \]
  \begin{itemize}
    \item The \textbf{IRF} is the sequence $\Phi_i$: a shock's echo over time.
    \item \textbf{Channel 1 -- lag order:} $\Phi_i$ comes from $A_1,\dots,A_{p_0}$,
          so the number of lags shapes the IRF.
    \item \textbf{Channel 2 -- identification:} $e_t$ mixes structural shocks;
          isolating one needs $e_t = B\varepsilon_t$ with $\Sigma = BB'$.
  \end{itemize}
  \begin{block}{Takeaway}
    An IRF depends on \emph{two} choices. We target lag uncertainty;
    set identification adds a second layer (next).
  \end{block}
\end{frame}

% ===== T2: sign restrictions, set identification, median target =====
\begin{frame}{Sign restrictions: identification without a single answer}
  \textbf{The problem:} infinitely many $B$ satisfy $\Sigma = BB'$
  --- so which structural story is right?
  \begin{itemize}
    \item \textbf{Cholesky:} imposes a strong assumption (a fixed ordering)
          to pin down \emph{one} answer.
    \item \textbf{Uhlig (2005) -- agnostic:} let the data speak; impose only a few
          sign restrictions everyone can agree on.
    \item Sign restrictions rule out the ``wrong-signed'' models and keep the rest
          $\Rightarrow$ they identify a \textbf{set} of admissible IRFs, not one.
  \end{itemize}
  \vspace{0.15cm}
  \begin{block}{Picking one representative}
    To compare against a single true IRF, we take the \textbf{median target}
    (Fry \& Pagan, 2011): the actual admissible model closest to the pointwise median.
  \end{block}
\end{frame}

% ===== T3: bias-variance trade-off and information criteria =====
\begin{frame}{The bias--variance trade-off}
  \textbf{An information criterion embodies the trade-off:}
  \[
    \mathrm{IC}(p) =
    \underbrace{\ln|\hat\Sigma(p)|}_{\text{penalizes underfitting}}
    \; + \;
    \underbrace{c_T\,\tfrac{K^2 p}{N}}_{\text{penalizes overfitting}}
    \qquad\quad
    \bigl(\text{MSE} = \text{Bias}^2 + \text{Variance}\bigr)
  \]
  \begin{itemize}
    \item Too few lags $\Rightarrow$ bias (underfitting); too many $\Rightarrow$ variance (overfitting).
    \item Criteria differ only in the penalty weight $c_T$:
          $\hat p_{BIC} \le \hat p_{HQC} \le \hat p_{AIC}$.
    \item AIC is the \textbf{most lenient} and stays closest to the true lag order.
    \item Caveat: AIC is usually seen as prone to \emph{overfitting}, but in a VAR each
          extra lag adds a full $(K\times K)$ block --- so here even AIC can \emph{underfit}.
  \end{itemize}
  \begin{block}{Key insight (Ivanov \& Kilian)}
    Recovering the \emph{true} lag order is not the goal --- minimizing MSE is.
    Sometimes an underfitting model wins.
  \end{block}
\end{frame}

% ===== T4a: Bayesian extension I -- BVAR shrinkage (solves curse of dimensionality) =====
\begin{frame}{Our first extension: BVAR shrinkage}
  \textbf{The frequentist problem --- curse of dimensionality:}
  \begin{itemize}
    \item A 12-lag system needs $K^2 p = 192$ coefficients against $\sim\!240$ observations;
          each estimated coefficient costs one degree of freedom.
  \end{itemize}
  \textbf{The shrinkage idea (Paper~I):} don't \emph{discard} lags --- \emph{discipline} them
  by pulling coefficients toward a prior.
  \[
    \text{posterior} \approx w\,(\text{data estimate}) + (1-w)\,(\text{prior mean})
  \]
  \begin{itemize}
    \item The tightness $\lambda_1$ sets how hard we shrink.
    \item Keep all 12 lags, but rein in the higher-order coefficients.
  \end{itemize}
  \begin{block}{What it solves}
    Shrinkage tames the \textbf{dimensionality} problem: many coefficients, few observations.
  \end{block}
\end{frame}

% ===== T4b: Bayesian extension II -- BMA (solves model uncertainty, NOT dimensionality) =====
\begin{frame}{Our second extension: Bayesian model averaging}
  \textbf{A different problem --- model uncertainty:}
  \begin{itemize}
    \item We do not know the true lag order. Committing to a single one ignores that
          uncertainty entirely.
  \end{itemize}
  \textbf{The averaging idea (Paper~II):} keep \emph{all} candidate lag orders and combine them.
  \[
    p(\Delta \mid y) = \sum_{k} p(\Delta \mid M_k, y)\, p(M_k \mid y)
  \]
  \begin{itemize}
    \item Each model $M_k$ (a lag order) is weighted by its posterior model probability.
    \item Instead of selecting one model, we \textbf{average} over many.
  \end{itemize}
  \begin{block}{What it solves}
    Averaging addresses \textbf{model uncertainty}.
  \end{block}
\end{frame}

\section{Results}

\subsection{Paper I: BVAR shrinkage}

\begin{frame}{Paper I: To Shrink or to Select?}
  \begin{itemize}
    \item Focus: comparison of information criteria and Bayesian VAR shrinkage.
    \item Main methods: AIC, BIC, HQC, and BVAR with different shrinkage hyperparameters $\lambda_1$.
    \item Main finding: AIC performs best among the classical criteria because it avoids severe underfitting.
    \item BVAR often outperforms AIC, especially when the true lag order is large.
  \end{itemize}
\end{frame}

\begin{frame}{Key results from Paper I}
  \begin{itemize}
    \item BIC and HQC are too parsimonious in finite samples and introduce substantial bias.
    \item The BVAR approach reduces variance without relying on a hard lag truncation.
    \item A moderate prior shrinkage is typically more useful than very tight shrinkage.
    \item Forecast-oriented tuning, such as MDD-based selection, does not necessarily optimize structural IRF recovery.
  \end{itemize}
\end{frame}

\begin{frame}{Takeaway from Paper I}
  \begin{block}{Conclusion}
    For structural inference, shrinkage can be more robust than selecting a single lag order when sample sizes are modest.
  \end{block}
\end{frame}

\subsection{Paper II: Bayesian model averaging}

\begin{frame}{Paper II: Hedging Against Lag Order Uncertainty}
  \begin{itemize}
    \item Focus: Bayesian model averaging as a strategy for dealing with model uncertainty.
    \item Main methods: averaging over lag orders and shrinkage specifications.
    \item Main finding: BMA reduces the sensitivity of IRF estimates to an uncertain lag order.
    \item The joint-grid BMA strategy often dominates the simple AIC benchmark.
  \end{itemize}
\end{frame}

\begin{frame}{Key results from Paper II}
  \begin{itemize}
    \item Averaging across specifications diversifies uncertainty rather than forcing a single choice.
    \item BMA performs especially well when the true lag order is high and the data are noisy.
    \item The method protects against overfitting and underfitting at the same time.
    \item Its success rests on weighting a broad set of plausible models rather than selecting one.
  \end{itemize}
\end{frame}

\begin{frame}{Takeaway from Paper II}
  \begin{block}{Conclusion}
    Model averaging is a natural response to lag-order uncertainty when the target is accurate structural inference.
  \end{block}
\end{frame}

\section{Conclusion}

\begin{frame}{How the two papers relate}
  \begin{table}
    \centering
    \begin{tabular}{p{0.28\textwidth}p{0.28\textwidth}p{0.28\textwidth}}
      \toprule
      \textbf{Dimension} & \textbf{Paper I} & \textbf{Paper II} \\
      \midrule
      Main tool & BVAR shrinkage & Bayesian model averaging \\
      Core idea & Regularize the VAR & Average over competing models \\
      Advantage & Reduces variance & Hedging against model uncertainty \\
      Main limitation & Prior choice matters & Computation and weighting matter \\
      \bottomrule
    \end{tabular}
  \end{table}
\end{frame}

\begin{frame}{Shared conclusion}
  \begin{itemize}
    \item In finite samples, hard lag selection can be too brittle.
    \item Shrinkage and averaging provide robust alternatives.
    \item The best strategy depends on the objective, but both papers support a more flexible treatment of lag uncertainty.
  \end{itemize}
\end{frame}

\begin{frame}{Concluding message}
  \begin{block}{Bottom line}
    For set-identified SVARs, the most reliable inference often comes from methods that acknowledge uncertainty rather than forcing a single lag order.
  \end{block}
  \vspace{0.3cm}
  \begin{itemize}
    \item AIC remains a strong benchmark.
    \item BVAR and BMA are attractive alternatives when structural IRFs are the target.
    \item Moderate shrinkage and model averaging are especially promising in practice.
  \end{itemize}
\end{frame}
